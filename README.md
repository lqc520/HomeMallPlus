```
  function getMyCookie(key) {
    var val = "";

    // 对cookie操作
    var cookies = document.cookie;
    cookies = cookies.replace(/\s/, "");
    var cookie_array = cookies.split(";");
    for (i = 0; i < cookie_array.length; i++) {
      // yh_mch=lilei
      var cookie = cookie_array[i];
      var array = cookie.split("=");
      if (array[0] == key) {
        val = array[1];
      }
    }

    return val;
  }
```
